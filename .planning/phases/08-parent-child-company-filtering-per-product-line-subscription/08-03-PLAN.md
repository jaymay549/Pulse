**Part A: Create `src/hooks/useActiveProductLine.tsx`**

This file provides a React Context for the active product line, fetches subscriptions via the `vendor_subscribed_slugs` RPC, and auto-selects the first product line alphabetically.

```typescript
import { createContext, useContext, useState, useEffect, type ReactNode } from "react";
import { useQuery } from "@tanstack/react-query";
import { vendorSupabase } from "@/integrations/supabase/vendorClient";

export interface ActiveProductLine {
  slug: string;
  name: string;
  tier: string;
}

interface ActiveProductLineContextValue {
  activeProductLine: ActiveProductLine | null;
  setActiveProductLine: (pl: ActiveProductLine) => void;
  subscriptions: ActiveProductLine[];
  isLoading: boolean;
}

const ActiveProductLineContext = createContext<ActiveProductLineContextValue>({
  activeProductLine: null,
  setActiveProductLine: () => {},
  subscriptions: [],
  isLoading: true,
});

interface ActiveProductLineProviderProps {
  isVendorAuth: boolean;
  vendorUserId: string | undefined;
  children: ReactNode;
}

export function ActiveProductLineProvider({ isVendorAuth, vendorUserId, children }: ActiveProductLineProviderProps) {
  const [activeProductLine, setActiveProductLine] = useState<ActiveProductLine | null>(null);

  const { data: subscriptions = [], isLoading } = useQuery({
    queryKey: ["vendor-product-subscriptions", vendorUserId],
    queryFn: async () => {
      const { data, error } = await vendorSupabase
        .rpc("vendor_subscribed_slugs" as never);
      if (error) throw error;
      return ((data ?? []) as Array<{ slug: string; tier: string; product_line_name: string }>)
        .map((row) => ({
          slug: row.slug,
          name: row.product_line_name,
          tier: row.tier,
        }));
    },
    enabled: isVendorAuth && !!vendorUserId,
  });

  // Auto-select first product line when subscriptions load (per D-09)
  useEffect(() => {
    if (subscriptions.length > 0 && !activeProductLine) {
      setActiveProductLine(subscriptions[0]); // Already sorted alphabetically by SQL
    }
  }, [subscriptions, activeProductLine]);

  return (
    <ActiveProductLineContext.Provider value={{ activeProductLine, setActiveProductLine, subscriptions, isLoading }}>
      {children}
    </ActiveProductLineContext.Provider>
  );
}

export function useActiveProductLine(): ActiveProductLineContextValue {
  return useContext(ActiveProductLineContext);
}
```

**Part B: Create `src/components/vendor-dashboard/VendorProductLineSwitcher.tsx`**

Compact header dropdown using shadcn/ui `Select`. Hidden when vendor has 0-1 subscriptions (per D-10 + Open Question 2 recommendation).

```typescript
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import { useActiveProductLine } from "@/hooks/useActiveProductLine";

export function VendorProductLineSwitcher() {
  const { activeProductLine, setActiveProductLine, subscriptions } = useActiveProductLine();

  // Hide if no choice to make (0 or 1 subscriptions)
  if (subscriptions.length <= 1) return null;

  return (
    <Select
      value={activeProductLine?.slug ?? ""}
      onValueChange={(slug) => {
        const found = subscriptions.find((s) => s.slug === slug);
        if (found) setActiveProductLine(found);
      }}
    >
      <SelectTrigger className="h-8 text-xs bg-white border-slate-200 text-slate-700 hover:bg-slate-50 w-auto min-w-[140px] max-w-[200px]">
        <SelectValue placeholder="Select product line" />
      </SelectTrigger>
      <SelectContent>
        {subscriptions.map((s) => (
          <SelectItem key={s.slug} value={s.slug}>
            {s.name}
          </SelectItem>
        ))}
      </SelectContent>
    </Select>
  );
}
```
